custom_gui_testing = {
	
	open_custom_gui_debug = {
		available = { always = yes }
		visible = { always = yes }
		fire_only_once = no
		cost = 0
		complete_effect = {
			if = {
				limit = { has_country_flag = custom_gui_open }
				clr_country_flag = custom_gui_open
			}
			else = {
				set_country_flag = custom_gui_open
			}
		}
		ai_will_do = { factor = 0 }
	}

	force_open_custom_gui_debug = {
		available = { always = yes }
		visible = { NOT = { has_country_flag = custom_gui_open } }
		fire_only_once = no
		cost = 0
		complete_effect = {
			set_country_flag = custom_gui_open
		}
		ai_will_do = { factor = 0 }
	}

	force_close_custom_gui_debug = {
		available = { always = yes }
		visible = { has_country_flag = custom_gui_open }
		fire_only_once = no
		cost = 0
		complete_effect = {
			clr_country_flag = custom_gui_open
		}
		ai_will_do = { factor = 0 }
	}
}
